function doPost(e) {
  // Set CORS headers for response
  const corsHeaders = {
    "Access-Control-Allow-Origin": "*",
    "Access-Control-Allow-Methods": "GET, POST, OPTIONS",
    "Access-Control-Allow-Headers": "Content-Type",
    "Access-Control-Max-Age": "86400"
  };
  
  try {
    let requestData;
    if (e.postData.type === "application/json") {
      requestData = JSON.parse(e.postData.contents);
    } else {
      requestData = JSON.parse(e.postData.getDataAsString());
    }
    
    const action = requestData.action;
    
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    
    let result;
    switch(action) {
      case "ADD_MASTER":
        result = handleAddMaster(ss, requestData);
        break;
      case "UPDATE_MASTER":
        result = handleUpdateMaster(ss, requestData);
        break;
      case "ADD_MEMBER":
        result = handleAddMember(ss, requestData);
        break;
      default:
        result = { status: "error", error: "Unknown action: " + action };
    }
    
    return createResponse(result, corsHeaders);
    
  } catch(error) {
    return createResponse({ status: "error", error: error.toString() }, corsHeaders);
  }
}

// ==================== HANDLE UPDATE MASTER ====================
function handleUpdateMaster(ss, data) {
  let sheet = ss.getSheetByName("Master");
  
  if (!sheet) {
    sheet = ss.insertSheet("Master");
    sheet.appendRow(["Branch ID", "Home ID", "Address Line 1", "Address Line 2", 
                     "Area", "Location", "Post Office", "District"]);
  }
  
  // Find and update existing record
  const rangeData = sheet.getDataRange().getValues();
  let found = false;
  
  for (let i = 1; i < rangeData.length; i++) {
    const row = rangeData[i];
    if (row[0] === data.branchId && row[1] === data.homeId) {
      const rowNumber = i + 1;
      // Update all columns
      if (data.addressLine1 !== undefined) sheet.getRange(rowNumber, 3).setValue(data.addressLine1);
      if (data.addressLine2 !== undefined) sheet.getRange(rowNumber, 4).setValue(data.addressLine2);
      if (data.area !== undefined) sheet.getRange(rowNumber, 5).setValue(data.area);
      if (data.location !== undefined) sheet.getRange(rowNumber, 6).setValue(data.location);
      if (data.postOffice !== undefined) sheet.getRange(rowNumber, 7).setValue(data.postOffice);
      if (data.district !== undefined) sheet.getRange(rowNumber, 8).setValue(data.district);
      found = true;
      break;
    }
  }
  
  if (!found) {
    // If not found, add as new
    sheet.appendRow([
      data.branchId || "",
      data.homeId || "",
      data.addressLine1 || "",
      data.addressLine2 || "",
      data.area || "",
      data.location || "",
      data.postOffice || "",
      data.district || ""
    ]);
  }
  
  return { 
    status: "success", 
    message: found ? "Master record updated successfully" : "Master record added successfully",
    branchId: data.branchId,
    homeId: data.homeId
  };
}

// ==================== CREATE RESPONSE ====================
function createResponse(data, corsHeaders) {
  const output = ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
  
  if (corsHeaders) {
    for (const key in corsHeaders) {
      output.setHeader(key, corsHeaders[key]);
    }
  }
  
  return output;
}
